Predicate logic

Formal language
  • Logical symbols: True, False, ∧, ∨, →, ∀, ∃.
  • Predicate symbols: P, Q, R, etc
    • Think of a predicate as a "parameterized" proposition, e.g.,
      • P(x,y) := "x > y"
      • Q(x) := "x is mortal"
      • R(x,y,z) := "x + y = z"
  • Variables: x, y, z, etc
    • variables denote individuals
  • Formulas:
    • True, False are formulas.
    • If R is a predicate symbol in n variables, then R x1 ... xn is a formula.
    • If P and Q are formulas, then
      • P ∧ Q
      • P ∨ Q
      • P → Q
      are formulas.
    • If P x is a formula and x is a variable of type A, then
      • ∀ x : A, P x
      • ∃ x : A, P x
      are formulas.

∀ (comp: "polymorphic" function, logic: for all)

  • Introduction (function abstraction):
    • Computation: Given x : A |- e : P x, we have fun x : A ⇒ e : ∀ x : A, P x.
    • Logic: To prove ∀ x : A, P x, assume an arbitrary x : A and show P x.
  • Elimination (function application):
    • Computation: Given f : ∀ x : A, P x and a : A, we have f a : P a.
    • Elimination: If we know ∀ x : A, P x, then P a holds for any a : A.

∃ (comp: "polymorphic" pair, logic: exists)

  • Introduction:
    • Computation: Given a : A and prf : P a, we have (a, prf) : ∃ x : A, P x.
    • Logic: To show ∃ x : A, P x, find an example a : A and a proof prf : P a.
  • Elimination:
    • Computation: Given pair : ∃ x : A, P x, we have pr1 pair : A and pr2 pair : P (pr1 pair).
    • Logic: If we know ∃ x : A, P x, then we know of an example a : A so that P a holds.

Theorem ex_then_and :
  ∀ (A : Set) (P Q : A → Prop),
  (∃ x : A, P x ∧ Q x) → (∃ x : A, P x) ∧ (∃ x : A, Q x).
Proof.
  intros A P Q [x [px qx]].
  split.
  - ∃ x.
    exact px.
  - ∃ x.
    exact qx.
Qed.

Theorem ex_then_or :
  ∀ (A : Set) (P Q : A → Prop),
    (∃ x : A, P x ∨ Q x) → (∃ x : A, P x) ∨ (∃ x : A, Q x).
Proof.
  intros A P Q [x [px | qx]].
  - left.
    ∃ x.
    exact px.
  - right.
    ∃ x.
    exact qx.
Qed.

Theorem fa_iff_and :
  ∀ (A : Set) (P Q : A → Prop),
    (∀ x : A, P x ∧ Q x) ↔ (∀ x : A, P x) ∧ (∀ x : A, Q x).
Proof.
  intros A P Q.
  split.
  - intros f.
    split.
    + intros x.
      destruct (f x) as [px _].
      exact px.
    + intros x.
      destruct (f x) as [_ qx].
      exact qx.
  - intros [f g] x.
    split.
    + apply f.
    + apply g.
Qed.

Theorem fa_dist_or :
  ∀ (A : Set) (P Q : A → Prop),
    (∀ x : A, P x) ∨ (∀ x : A, Q x) → (∀ x : A, P x ∨ Q x).
Proof.
  intros A P Q [f | g] x.
  - left.
    apply f.
  - right.
    apply g.
Qed.

Theorem fa_dist_imp :
  ∀ (A : Set) (P Q : A → Prop),
    (∀ x : A, P x → Q x) → (∀ x : A, P x) → (∀ x : A, Q x).
Proof.
  intros A P Q f g x.
  apply f.
  apply g.
Qed.

Theorem ex_then_not_fa_not :
  ∀ (A : Set) (P : A → Prop),
    (∃ x : A, P x) → (~ (∀ x : A, ¬ P x)).
Proof.
  intros A P [x px] f.
  apply (f x).
  exact px.
Qed.

Section ClassicalLogic.
Definition lem :=
  ∀ P : Prop, ¬ P ∨ P.
Definition peirce :=
  ∀ P Q : Prop, ((P → Q) → P) → P.
Definition dne :=
  ∀ P : Prop, ¬ ¬ P → P.
Definition neg_demorgan :=
  ∀ P Q : Prop, ¬ (~ P ∧ ¬ Q) → P ∨ Q.
Definition classical_implies :=
  ∀ P Q : Prop, (P → Q) → ¬ P ∨ Q.

Theorem lem_then_peirce :
  lem → peirce.
Proof.
  unfold lem.
  intros lem P Q f.
  destruct (lem P) as [np | p].
  - apply f.
    intros p.
    contradiction.
  - exact p.
Qed.

Theorem peirce_then_dne :
  peirce → dne.
Proof.
  unfold peirce.
  intros peirce.
  intros P nnp.
  apply peirce with (Q := False).
  intros np.
  contradiction.
Qed.

Lemma nonrefutable :
  ∀ P : Prop, ¬ ¬ (~ P ∨ P).
Proof.
  intros P n.
  apply n.
  left.
  intros p.
  apply n.
  right.
  exact p.
Qed.

Theorem dne_then_lem :
  dne → lem.
Proof.
  unfold dne.
  intros dne P.
  apply dne.
  apply nonrefutable.
Qed.

Theorem dne_then_neg_demorgan :
  dne → neg_demorgan.
Proof.
  unfold dne.
  intros dne P Q f.
  set (h := dne_then_lem dne).
  destruct (h P) as [np | p].
  - right.
    destruct (h Q) as [nq | q].
    + set (h2 := conj np nq).
      contradiction.
    + exact q.
  - left.
    exact p.
Qed.

Theorem neg_demorgan_then_classical_implies :
  neg_demorgan → classical_implies.
Proof.
  unfold neg_demorgan.
  intros neg_demorgan P Q f.
  apply neg_demorgan.
  intros [nnp nq].
  apply nnp.
  intros p.
  apply nq.
  apply f.
  exact p.
Qed.

Theorem classical_implies_then_lem :
  classical_implies → lem.
Proof.
  unfold classical_implies.
  intros classical_implies P.
  apply classical_implies.
  intros p.
  exact p.
Qed.

End ClassicalLogic.
